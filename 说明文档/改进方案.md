**A. 多径衰落信道**
```python
# 新增到 dsp_engine.py
@staticmethod
def channel_rayleigh_fading(symbols, snr_db, doppler_hz=10, sample_rate=1e6):
    """瑞利衰落信道（无直射路径，如城市环境）"""
    # 生成时变衰落系数
    num_samples = len(symbols)
    fd_norm = doppler_hz / sample_rate  # 归一化多普勒频率

    # Jakes模型生成衰落
    h = np.zeros(num_samples, dtype=complex)
    N = 16  # 散射路径数
    for n in range(N):
        phase = 2 * np.pi * np.random.rand()
        angle = 2 * np.pi * n / N
        h += np.exp(1j * (2 * np.pi * fd_norm * np.arange(num_samples) * np.cos(angle) + phase))
    h = h / np.sqrt(N)

    # 应用衰落
    faded_symbols = symbols * h

    # 添加AWGN
    return DSPEngine.channel_awgn(faded_symbols, snr_db), h

@staticmethod
def channel_rician_fading(symbols, snr_db, K_factor_db=10, doppler_hz=5):
    """莱斯衰落信道（有直射路径，如卫星通信）"""
    K = 10 ** (K_factor_db / 10)  # 莱斯因子
    # 直射分量 + 散射分量
    los_component = np.sqrt(K / (K + 1))
    scatter_component = np.sqrt(1 / (K + 1))

    # 生成散射衰落（类似瑞利）
    h_scatter = np.random.randn(len(symbols)) + 1j * np.random.randn(len(symbols))
    h_scatter = h_scatter / np.sqrt(2)

    h = los_component + scatter_component * h_scatter
    faded_symbols = symbols * h

    return DSPEngine.channel_awgn(faded_symbols, snr_db), h
```

**B. 自由空间路径损耗（精确计算）**
```python
@staticmethod
def calculate_fspl(distance_km, frequency_ghz):
    """
    自由空间路径损耗
    FSPL(dB) = 20*log10(d) + 20*log10(f) + 92.45
    d: 距离(km), f: 频率(GHz)
    """
    if distance_km <= 0 or frequency_ghz <= 0:
        return 0
    fspl_db = 20 * np.log10(distance_km) + 20 * np.log10(frequency_ghz) + 92.45
    return fspl_db

@staticmethod
def calculate_link_budget(tx_power_dbm, tx_gain_dbi, rx_gain_dbi,
                         distance_km, frequency_ghz,
                         atmospheric_loss_db=0, rain_loss_db=0):
    """
    完整链路预算计算
    返回接收功率(dBm)和SNR估算
    """
    fspl = DSPEngine.calculate_fspl(distance_km, frequency_ghz)
    rx_power_dbm = (tx_power_dbm + tx_gain_dbi + rx_gain_dbi
                    - fspl - atmospheric_loss_db - rain_loss_db)

    # 假设接收机噪声系数 = 3dB, 带宽 = 1MHz
    noise_floor_dbm = -174 + 10 * np.log10(1e6) + 3  # ≈ -111 dBm
    snr_db = rx_power_dbm - noise_floor_dbm

    return rx_power_dbm, snr_db
```

```python
def modulate(bits, mod_type):

        if len(bits) % 2 != 0: bits = np.append(bits, 0)
```

**B. OFDM（正交频分复用）**
```python
@staticmethod
def modulate_ofdm(bits, mod_type="QPSK", n_subcarriers=64, cp_length=16):
    """
    OFDM调制
    n_subcarriers: 子载波数量
    cp_length: 循环前缀长度
    """
    # 1. 串并转换和子载波调制
    symbols = DSPEngine.modulate(bits, mod_type)

    # 2. 补齐到子载波数量的倍数
    n_symbols = len(symbols)
    n_ofdm_symbols = int(np.ceil(n_symbols / n_subcarriers))
    padded_len = n_ofdm_symbols * n_subcarriers
    if n_symbols < padded_len:
        symbols = np.concatenate([symbols, np.zeros(padded_len - n_symbols)])

    # 3. 重塑为OFDM符号矩阵
    symbols_matrix = symbols.reshape(n_ofdm_symbols, n_subcarriers)

    # 4. IFFT（将频域转时域）
    time_domain = np.fft.ifft(symbols_matrix, axis=1)

    # 5. 添加循环前缀（抗多径）
    cp = time_domain[:, -cp_length:]
    ofdm_symbols = np.concatenate([cp, time_domain], axis=1)

    return ofdm_symbols.flatten()
```

```python
@staticmethod
def carrier_frequency_offset(symbols, cfo_hz, sample_rate):
    """模拟载波频偏"""
    t = np.arange(len(symbols)) / sample_rate
    return symbols * np.exp(1j * 2 * np.pi * cfo_hz * t)

@staticmethod
def estimate_and_correct_cfo(rx_symbols, sample_rate):
    """估计并校正载波频偏（简化算法）"""
    # 使用自相关方法估计频偏
    delay = 16
    R = np.sum(rx_symbols[:-delay] * np.conj(rx_symbols[delay:]))
    cfo_estimate = np.angle(R) * sample_rate / (2 * np.pi * delay)

    # 校正
    t = np.arange(len(rx_symbols)) / sample_rate
    corrected = rx_symbols * np.exp(-1j * 2 * np.pi * cfo_estimate * t)

    return corrected, cfo_estimate
```

### 2.1 交织器（Interleaver）

#### 作用
对抗突发错误，将连续的比特分散到不同时间/频率位置

```python
@staticmethod
def interleave(bits, method="block", rows=8, cols=None):
    """
    交织器
    method: "block" (块交织) 或 "convolutional" (卷积交织)
    """
    if method == "block":
        if cols is None:
            cols = int(np.ceil(len(bits) / rows))

        # 补齐
        total = rows * cols
        if len(bits) < total:
            bits = np.concatenate([bits, np.zeros(total - len(bits), dtype=int)])

        # 按行写入，按列读出
        matrix = bits[:total].reshape(rows, cols)
        interleaved = matrix.T.flatten()
        return interleaved

    return bits

@staticmethod
def deinterleave(bits, method="block", rows=8, cols=None):
    """解交织"""
    if method == "block":
        if cols is None:
            cols = int(np.ceil(len(bits) / rows))

        total = rows * cols
        if len(bits) < total:
            bits = np.concatenate([bits, np.zeros(total - len(bits), dtype=int)])

        # 按列写入，按行读出
        matrix = bits[:total].reshape(cols, rows).T
        deinterleaved = matrix.flatten()
        return deinterleaved

    return bits
```

### 2.2 CRC校验

```python
@staticmethod
def calculate_crc(bits, polynomial=0x1021, init=0xFFFF):
    """
    CRC-16-CCITT校验
    polynomial: 生成多项式 (默认0x1021)
    """
    crc = init
    for bit in bits:
        crc ^= (int(bit) << 15)
        for _ in range(1):
            if crc & 0x8000:
                crc = (crc << 1) ^ polynomial
            else:
                crc = crc << 1
            crc &= 0xFFFF
    return crc

@staticmethod
def add_crc(bits, crc_bits=16):
    """在数据后添加CRC校验码"""
    crc = DSPEngine.calculate_crc(bits)
    crc_array = np.array([int(b) for b in format(crc, f'0{crc_bits}b')], dtype=int)
    return np.concatenate([bits, crc_array])

@staticmethod
def check_crc(bits_with_crc, crc_bits=16):
    """检查CRC，返回(数据, 是否正确)"""
    if len(bits_with_crc) < crc_bits:
        return bits_with_crc, False

    data = bits_with_crc[:-crc_bits]
    received_crc = bits_with_crc[-crc_bits:]

    calculated_crc = DSPEngine.calculate_crc(data)
    calculated_crc_array = np.array([int(b) for b in format(calculated_crc, f'0{crc_bits}b')], dtype=int)

    is_correct = np.array_equal(received_crc, calculated_crc_array)
    return data, is_correct
```

### 2.4 Turbo码

```python
@staticmethod
def encode_turbo(bits, constraint_length=3):
    """
    Turbo码编码（双并行级联卷积码）
    """
    # 组成部分：
    # 1. 系统比特
    # 2. 第一个RSC编码器输出
    # 3. 交织后的第二个RSC编码器输出

    # 简化实现
    systematic = bits
    parity1 = DSPEngine._rsc_encode(bits, constraint_length)
    interleaved = DSPEngine.interleave(bits, method="block")
    parity2 = DSPEngine._rsc_encode(interleaved, constraint_length)

    # 组合输出
    encoded = np.concatenate([systematic, parity1, parity2])
    return encoded

@staticmethod
def _rsc_encode(bits, constraint_length=3):
    """递归系统卷积(RSC)编码器"""
    # 简化实现
    state = 0
    output = []
    for bit in bits:
        # 状态转移和输出计算（简化）
        output_bit = (bit + state) % 2
        output.append(output_bit)
        state = (state + bit) % 2
    return np.array(output, dtype=int)
```

### 3.1 动态路由算法

#### 当前问题
- 玩家手动选择路径，缺乏自动路由
- 无法应对链路质量动态变化

#### 改进方案

**A. Dijkstra最短路径（基于SNR权重）**
```python
class NetworkRouter:
    @staticmethod
    def dijkstra_routing(nodes, snr_matrix, src_idx, dest_idx):
        """
        基于SNR的最短路径算法
        权重 = -SNR (SNR越高，权重越低)
        """
        n = len(nodes)
        # 将SNR转换为权重（SNR越高越好，所以取负值）
        weights = -snr_matrix.copy()
        weights[weights == 0] = 999  # 无连接

        dist = [float('inf')] * n
        dist[src_idx] = 0
        visited = [False] * n
        parent = [-1] * n

        for _ in range(n):
            # 找未访问的最小距离节点
            u = -1
            min_dist = float('inf')
            for i in range(n):
                if not visited[i] and dist[i] < min_dist:
                    min_dist = dist[i]
                    u = i

            if u == -1 or u == dest_idx:
                break

            visited[u] = True

            # 更新邻居距离
            for v in range(n):
                if not visited[v] and weights[u][v] < 999:
                    new_dist = dist[u] + weights[u][v]
                    if new_dist < dist[v]:
                        dist[v] = new_dist
                        parent[v] = u

        # 重建路径
        if dist[dest_idx] == float('inf'):
            return None  # 无路径

        path = []
        current = dest_idx
        while current != -1:
            path.append(current)
            current = parent[current]
        path.reverse()

        return path

    @staticmethod
    def calculate_path_quality(path, snr_matrix):
        """计算路径的总体质量（最弱链路SNR）"""
        if len(path) < 2:
            return 0

        min_snr = float('inf')
        for i in range(len(path) - 1):
            snr = snr_matrix[path[i]][path[i+1]]
            min_snr = min(min_snr, snr)

        return min_snr
```

**B. 链路状态协议（OSPF简化版）**
```python
class LinkStateProtocol:
    def __init__(self, nodes, snr_matrix):
        self.nodes = nodes
        self.snr_matrix = snr_matrix
        self.link_state_db = {}  # 链路状态数据库
        self.update_interval = 5.0  # 秒

    def broadcast_link_state(self, node_idx):
        """节点广播其链路状态"""
        neighbors = []
        for i, snr in enumerate(self.snr_matrix[node_idx]):
            if snr > -10:  # 可用链路阈值
                neighbors.append({"node": i, "cost": -snr})

        self.link_state_db[node_idx] = {
            "neighbors": neighbors,
            "timestamp": time.time()
        }

    def compute_routing_table(self, src_idx):
        """基于链路状态数据库计算路由表"""
        # 使用Dijkstra算法
        routing_table = {}
        for dest_idx in range(len(self.nodes)):
            if dest_idx != src_idx:
                path = NetworkRouter.dijkstra_routing(
                    self.nodes, self.snr_matrix, src_idx, dest_idx
                )
                routing_table[dest_idx] = path

        return routing_table
```

### 3.2 拥塞控制

```python
class CongestionControl:
    def __init__(self, initial_window=1, max_window=64):
        self.cwnd = initial_window  # 拥塞窗口
        self.ssthresh = max_window // 2  # 慢启动阈值
        self.max_window = max_window

    def on_ack_received(self):
        """收到ACK，增加窗口"""
        if self.cwnd < self.ssthresh:
            # 慢启动阶段：指数增长
            self.cwnd = min(self.cwnd * 2, self.max_window)
        else:
            # 拥塞避免阶段：线性增长
            self.cwnd = min(self.cwnd + 1, self.max_window)

    def on_packet_loss(self):
        """检测到丢包，减小窗口"""
        self.ssthresh = max(self.cwnd // 2, 1)
        self.cwnd = 1  # 重新慢启动

    def get_window_size(self):
        return int(self.cwnd)
```

```python
@staticmethod
def multipath_routing(nodes, snr_matrix, src_idx, dest_idx, num_paths=2):
    """
    多路径路由（负载均衡）
    返回多条不相交或部分重叠的路径
    """
    paths = []
    temp_matrix = snr_matrix.copy()

    for _ in range(num_paths):
        path = NetworkRouter.dijkstra_routing(nodes, temp_matrix, src_idx, dest_idx)
        if path is None:
            break

        paths.append(path)

        # 降低已使用链路的权重，鼓励使用其他路径
        for i in range(len(path) - 1):
            temp_matrix[path[i]][path[i+1]] -= 5

    return paths
```

### 4.1 帧结构设计

```python
class Frame:
    """通信帧结构"""
    def __init__(self):
        self.preamble = np.array([1,0,1,0,1,0,1,0], dtype=int)  # 前导码（用于同步）
        self.sync_word = np.array([1,1,1,0,0,0,1,0], dtype=int)  # 同步字
        self.header = None  # 帧头（包含长度、序号等）
        self.payload = None  # 有效载荷
        self.crc = None  # 校验码

    def build_frame(self, data_bits, frame_id=0):
        """构建完整帧"""
        # 帧头：8位帧ID + 16位长度
        header_bits = np.concatenate([
            np.array([int(b) for b in format(frame_id, '08b')], dtype=int),
            np.array([int(b) for b in format(len(data_bits), '016b')], dtype=int)
        ])

        # 组装帧
        frame_without_crc = np.concatenate([
            self.preamble,
            self.sync_word,
            header_bits,
            data_bits
        ])

        # 添加CRC
        crc = DSPEngine.calculate_crc(frame_without_crc)
        crc_bits = np.array([int(b) for b in format(crc, '016b')], dtype=int)

        complete_frame = np.concatenate([frame_without_crc, crc_bits])
        return complete_frame

    @staticmethod
    def parse_frame(rx_bits):
        """解析接收到的帧"""
        # 查找同步字
        sync_word = np.array([1,1,1,0,0,0,1,0], dtype=int)
        preamble_len = 8

        # 简化：假设已同步，直接解析
        if len(rx_bits) < preamble_len + 8 + 24 + 16:
            return None, False

        offset = preamble_len + 8  # 跳过前导码和同步字

        # 解析帧头
        frame_id_bits = rx_bits[offset:offset+8]
        frame_id = int(''.join(map(str, frame_id_bits)), 2)

        length_bits = rx_bits[offset+8:offset+24]
        payload_length = int(''.join(map(str, length_bits)), 2)

        # 提取载荷
        payload_start = offset + 24
        payload_end = payload_start + payload_length
        if payload_end + 16 > len(rx_bits):
            return None, False

        payload = rx_bits[payload_start:payload_end]

        # 校验CRC
        frame_with_crc = rx_bits[:payload_end+16]
        _, crc_ok = DSPEngine.check_crc(frame_with_crc, crc_bits=16)

        return {"frame_id": frame_id, "payload": payload}, crc_ok
```

```python
class ARQProtocol:
    """停等ARQ协议"""
    def __init__(self, max_retries=3, timeout=1.0):
        self.max_retries = max_retries
        self.timeout = timeout
        self.seq_num = 0

    def send_with_arq(self, data_bits, channel_func):
        """
        带ARQ的发送
        channel_func: 信道传输函数
        返回：(成功/失败, 重传次数)
        """
        retries = 0
        frame = Frame().build_frame(data_bits, frame_id=self.seq_num)

        while retries <= self.max_retries:
            # 发送帧
            rx_bits = channel_func(frame)

            # 解析接收
            parsed, crc_ok = Frame.parse_frame(rx_bits)

            if crc_ok and parsed is not None:
                self.seq_num = (self.seq_num + 1) % 256
                return True, retries

            retries += 1

        return False, retries


class HARQProtocol:
    """混合ARQ（结合FEC和ARQ）"""
    def __init__(self, max_retries=3):
        self.max_retries = max_retries
        self.soft_buffer = []  # 存储软信息用于合并

    def send_with_harq(self, data_bits, channel_func, code_type="Polar(128,64)"):
        """
        HARQ Type II（增量冗余）
        第一次发送：系统位
        重传：额外的校验位
        """
        # 编码
        encoded = DSPEngine.encode_data(data_bits, code_type)

        # 分割为系统位和校验位
        k = len(data_bits)
        systematic = encoded[:k]
        parity = encoded[k:]

        retries = 0
        self.soft_buffer = []

        # 第一次发送系统位
        rx_llr = channel_func(systematic, return_llr=True)
        self.soft_buffer.append(rx_llr)

        decoded = DSPEngine.decode_data(rx_llr, code_type, orig_len=k)
        _, crc_ok = DSPEngine.check_crc(decoded)

        if crc_ok:
            return True, retries

        # 重传校验位
        while retries < self.max_retries:
            retries += 1
            rx_parity_llr = channel_func(parity, return_llr=True)

            # 合并软信息
            combined_llr = np.concatenate([self.soft_buffer[0], rx_parity_llr])

            decoded = DSPEngine.decode_data(combined_llr, code_type, orig_len=k)
            _, crc_ok = DSPEngine.check_crc(decoded)

            if crc_ok:
                return True, retries

        return False, retries
```

```python
class SatelliteOrbit:
    """卫星轨道模拟"""
    def __init__(self, altitude_km, inclination_deg=0):
        self.altitude = altitude_km
        self.inclination = inclination_deg
        self.earth_radius = 6371  # km
        self.orbital_radius = self.earth_radius + altitude_km

        # 计算轨道周期（开普勒第三定律）
        mu = 398600.4418  # 地球引力常数 km³/s²
        self.period = 2 * np.pi * np.sqrt(self.orbital_radius**3 / mu)

    def get_position(self, time_sec):
        """获取卫星位置（简化为圆轨道）"""
        angle = 2 * np.pi * time_sec / self.period
        x = self.orbital_radius * np.cos(angle)
        y = self.orbital_radius * np.sin(angle)
        z = 0  # 简化为赤道轨道
        return np.array([x, y, z])

    def calculate_distance(self, pos1, pos2):
        """计算两点间距离"""
        return np.linalg.norm(pos1 - pos2)

    def calculate_doppler_shift(self, velocity_km_s, frequency_hz):
        """计算多普勒频移"""
        c = 299792.458  # 光速 km/s
        return frequency_hz * velocity_km_s / c

    def is_line_of_sight(self, sat_pos, ground_pos):
        """判断是否有视线（考虑地球遮挡）"""
        # 简化：检查仰角是否大于0度
        vec = sat_pos - ground_pos
        elevation = np.arcsin(vec[2] / np.linalg.norm(vec))
        return elevation > 0
```

```python
class AdaptiveModulation:
    """自适应调制编码（AMC）"""
    def __init__(self):
        # MCS表：(调制, 编码, 最低SNR要求)
        self.mcs_table = [
            ("BPSK", "Repetition(3,1)", -2),
            ("BPSK", "Hamming(7,4)", 2),
            ("QPSK", "Hamming(7,4)", 5),
            ("QPSK", "Polar(128,64)", 8),
            ("16QAM", "Polar(256,192)", 12),
            ("64QAM", "Polar(512,384)", 18),
        ]

    def select_mcs(self, current_snr_db):
        """根据当前SNR选择最优MCS"""
        best_mcs = self.mcs_table[0]
        for mcs in self.mcs_table:
            if current_snr_db >= mcs[2]:
                best_mcs = mcs
            else:
                break
        return best_mcs

    def calculate_throughput(self, mcs, bandwidth_hz):
        """计算吞吐量"""
        mod, code, _ = mcs

        # 符号速率 = 带宽
        symbol_rate = bandwidth_hz

        # 每符号比特数
        bits_per_symbol = {"BPSK": 1, "QPSK": 2, "16QAM": 4, "64QAM": 6}[mod]

        # 编码率
        code_rate = {"Repetition(3,1)": 1/3, "Hamming(7,4)": 4/7,
                     "Polar(128,64)": 0.5, "Polar(256,192)": 0.75,
                     "Polar(512,384)": 0.75}[code]

        throughput = symbol_rate * bits_per_symbol * code_rate
        return throughput
```

```python
class SpectrumAnalyzer:
    """频谱分析器"""
    @staticmethod
    def plot_spectrum(signal, sample_rate, title="Power Spectral Density"):
        """
        绘制功率谱密度
        """
        import matplotlib.pyplot as plt

        # 计算FFT
        fft_result = np.fft.fftshift(np.fft.fft(signal))
        freq = np.fft.fftshift(np.fft.fftfreq(len(signal), 1/sample_rate))

        # 功率谱密度（dB）
        psd = 10 * np.log10(np.abs(fft_result)**2 + 1e-12)

        plt.figure(figsize=(10, 6))
        plt.plot(freq / 1e6, psd)  # 频率单位：MHz
        plt.xlabel('Frequency (MHz)')
        plt.ylabel('PSD (dB)')
        plt.title(title)
        plt.grid(True)

        return plt

    @staticmethod
    def calculate_bandwidth(signal, sample_rate, threshold_db=-3):
        """
        计算-3dB带宽
        """
        fft_result = np.fft.fftshift(np.fft.fft(signal))
        psd = 10 * np.log10(np.abs(fft_result)**2 + 1e-12)

        max_psd = np.max(psd)
        threshold = max_psd + threshold_db

        # 找到超过阈值的频率范围
        above_threshold = psd > threshold
        indices = np.where(above_threshold)[0]

        if len(indices) > 0:
            bandwidth = (indices[-1] - indices[0]) * sample_rate / len(signal)
            return bandwidth
        return 0
```
