# 发射因果链动画实施方案

## 🎯 设计目标

在玩家点击"发送"后，在传输开始前，播放一段动画展示：
- 各个技术选择如何影响最终SNR
- 每个环节的具体数值变化
- 最终预期结果

**时机**：点击发送 → 因果链动画（2-3秒）→ 开始传输

---

## 🎬 动画流程设计

### 阶段1：淡入（0.3秒）

```
屏幕半透明遮罩淡入
因果链面板从中央展开
```

### 阶段2：逐步展示（2秒）

```
基础SNR节点 → 飞入 → 显示数值
  ↓ 箭头动画
调制影响节点 → 飞入 → 显示数值和变化
  ↓ 箭头动画
编码影响节点 → 飞入 → 显示数值和变化
  ↓ 箭头动画
天气影响节点 → 飞入 → 显示数值和变化
  ↓ 箭头动画
最终SNR节点 → 飞入 → 显示数值（高亮）
  ↓
预期结果 → 显示成功率和评级
```

### 阶段3：淡出（0.5秒）

```
面板缩小淡出
开始传输
```

```python
# causal_chain_animation.py（新文件）

import pygame
import numpy as np

class CausalChainAnimation:
    """因果链动画"""
    def __init__(self, config):
        self.config = config
        self.state = "idle"  # idle, playing, finished
        self.start_time = 0
        self.duration = 2.5  # 总时长2.5秒

        # 计算因果链数据
        self.chain_data = self.calculate_chain()

        # 节点位置
        self.nodes = []
        self.setup_nodes()

    def calculate_chain(self):
        """计算因果链各环节数据"""
        base_snr = self.config["base_snr"]

        # 调制影响
        mod_effect = self.get_modulation_effect(self.config["modulation"])
        snr_after_mod = base_snr + mod_effect

        # 编码影响
        code_effect = self.get_coding_effect(self.config["coding"])
        snr_after_code = snr_after_mod + code_effect

        # 天气影响
        weather_effect = self.get_weather_effect(self.config["weather"])
        final_snr = snr_after_code + weather_effect

        # 协议影响（如果有）
        protocol_effect = self.get_protocol_effect(self.config.get("protocol", "udp"))

        return {
            "base": {"snr": base_snr, "label": "基础SNR"},
            "modulation": {
                "snr": snr_after_mod,
                "effect": mod_effect,
                "label": self.config["modulation"],
                "icon": "📡"
            },
            "coding": {
                "snr": snr_after_code,
                "effect": code_effect,
                "label": self.config["coding"],
                "icon": "🔐"
            },
            "weather": {
                "snr": final_snr,
                "effect": weather_effect,
                "label": self.config["weather"],
                "icon": self.get_weather_icon(self.config["weather"])
            },
            "final": {
                "snr": final_snr,
                "success_rate": self.calculate_success_rate(final_snr),
                "label": "最终结果"
            }
        }

    def get_modulation_effect(self, modulation):
        """获取调制方式的SNR影响"""
        effects = {
            "BPSK": 0,      # 基准
            "QPSK": -2,     # 速度快但SNR略降
            "16QAM": -4,    # 速度更快但SNR降更多
        }
        return effects.get(modulation, 0)

    def get_coding_effect(self, coding):
        """获取编码方式的SNR增益"""
        if "Polar" in coding:
            return 3
        elif "Hamming" in coding:
            return 2
        elif "Repetition" in coding:
            return 2.5
        else:
            return 0

    def get_weather_effect(self, weather):
        """获取天气影响"""
        effects = {
            "clear": 0,
            "cloudy": -1,
            "rain": -3,
            "storm": -6,
            "solar_flare": -4
        }
        return effects.get(weather, 0)

    def get_weather_icon(self, weather):
        """获取天气图标"""
        icons = {
            "clear": "☀️",
            "cloudy": "☁️",
            "rain": "🌧️",
            "storm": "⛈️",
            "solar_flare": "☀️💥"
        }
        return icons.get(weather, "☁️")

    def calculate_success_rate(self, final_snr):
        """计算成功率"""
        if final_snr > 12:
            return 0.95
        elif final_snr > 8:
            return 0.80
        elif final_snr > 5:
            return 0.60
        else:
            return 0.30
```

```python
    def setup_nodes(self):
        """设置节点位置"""
        center_x = WINDOW_WIDTH // 2
        center_y = WINDOW_HEIGHT // 2
        spacing = 180

        self.nodes = [
            {"key": "base", "pos": (center_x - spacing * 2, center_y), "alpha": 0},
            {"key": "modulation", "pos": (center_x - spacing, center_y), "alpha": 0},
            {"key": "coding", "pos": (center_x, center_y), "alpha": 0},
            {"key": "weather", "pos": (center_x + spacing, center_y), "alpha": 0},
            {"key": "final", "pos": (center_x + spacing * 2, center_y), "alpha": 0}
        ]

    def start(self):
        """开始动画"""
        self.state = "playing"
        self.start_time = pygame.time.get_ticks()

    def update(self):
        """更新动画状态"""
        if self.state != "playing":
            return

        elapsed = (pygame.time.get_ticks() - self.start_time) / 1000.0

        # 逐个节点飞入
        for i, node in enumerate(self.nodes):
            node_start_time = i * 0.4  # 每个节点间隔0.4秒
            if elapsed > node_start_time:
                progress = min((elapsed - node_start_time) / 0.3, 1.0)
                node["alpha"] = int(255 * progress)

        # 动画结束
        if elapsed > self.duration:
            self.state = "finished"

    def draw(self, surface):
        """绘制动画"""
        if self.state == "idle":
            return

        # 半透明遮罩
        overlay = pygame.Surface((WINDOW_WIDTH, WINDOW_HEIGHT), pygame.SRCALPHA)
        overlay.fill((0, 0, 0, 180))
        surface.blit(overlay, (0, 0))

        # 绘制节点
        for i, node in enumerate(self.nodes):
            if node["alpha"] > 0:
                self.draw_node(surface, node, i)

        # 绘制箭头
        for i in range(len(self.nodes) - 1):
            if self.nodes[i]["alpha"] > 200 and self.nodes[i+1]["alpha"] > 0:
                self.draw_arrow(surface, self.nodes[i], self.nodes[i+1])

    def draw_node(self, surface, node, index):
        """绘制单个节点"""
        data = self.chain_data[node["key"]]
        x, y = node["pos"]
        alpha = node["alpha"]

        # 节点圆圈
        radius = 50
        node_surface = pygame.Surface((radius*2, radius*2), pygame.SRCALPHA)

        # 根据效果决定颜色
        if node["key"] == "base":
            color = (100, 100, 100, alpha)
        elif node["key"] == "final":
            snr = data["snr"]
            if snr > 10:
                color = (0, 255, 0, alpha)
            elif snr > 5:
                color = (255, 200, 0, alpha)
            else:
                color = (255, 100, 100, alpha)
        else:
            effect = data.get("effect", 0)
            if effect > 0:
                color = (0, 200, 100, alpha)
            elif effect < 0:
                color = (200, 100, 0, alpha)
            else:
                color = (100, 100, 100, alpha)

        pygame.draw.circle(node_surface, color, (radius, radius), radius)
        pygame.draw.circle(node_surface, (255, 255, 255, alpha), (radius, radius), radius, 3)
        surface.blit(node_surface, (x - radius, y - radius))

        surface.blit(snr_text, (x - snr_text.get_width()//2, y - 5))

        # 变化量

    def draw_arrow(self, surface, node1, node2):
        """绘制箭头"""
        x1, y1 = node1["pos"]
        x2, y2 = node2["pos"]
        alpha = min(node1["alpha"], node2["alpha"])

        # 箭头线
        pygame.draw.line(surface, (0, 180, 255, alpha),
                        (x1 + 50, y1), (x2 - 50, y2), 4)

        # 箭头头部
        arrow_size = 15
        angle = 0  # 水平箭头
        pygame.draw.polygon(surface, (0, 180, 255, alpha), [
            (x2 - 50, y2),
            (x2 - 50 - arrow_size, y2 - arrow_size//2),
            (x2 - 50 - arrow_size, y2 + arrow_size//2)
        ])

    def is_finished(self):
        """动画是否结束"""
        return self.state == "finished"
```
