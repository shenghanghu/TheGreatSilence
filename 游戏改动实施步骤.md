```python
# budget_system.py（新文件）

class BudgetManager:
    """预算管理系统"""
    def __init__(self, initial_budget=1000):
        self.total_budget = initial_budget  # 总预算
        self.current_budget = initial_budget  # 当前可用预算
        self.spent_this_level = 0  # 本关消耗
        self.transaction_history = []  # 交易历史

    def can_afford(self, cost):
        """检查是否有足够预算"""
        return self.current_budget >= cost

    def spend(self, cost, description=""):
        """消耗预算"""
        if not self.can_afford(cost):
            return False, "预算不足"

        self.current_budget -= cost
        self.spent_this_level += cost
        self.transaction_history.append({
            "type": "spend",
            "amount": cost,
            "description": description,
            "balance": self.current_budget
        })
        return True, f"消耗 {cost} 预算"

    def earn(self, amount, description=""):
        """获得预算"""
        self.current_budget += amount
        self.transaction_history.append({
            "type": "earn",
            "amount": amount,
            "description": description,
            "balance": self.current_budget
        })
        return True, f"获得 {amount} 预算"

    def reset_level(self):
        """重置关卡消耗统计"""
        self.spent_this_level = 0
```

```python
COST_TABLE = {

def calculate_transmission_cost(power_dbm, protocol="udp"):
    """计算传输成本"""
    base = COST_TABLE["transmission_base"]
    power_cost = (power_dbm - 30) * COST_TABLE["power_multiplier"]
```

```python
                    "ber_multiplier": 1.5  # BER增加50%
                }
            },
            "solar_flare": {
                "name": "太阳耀斑",
                "icon": "picture/weather_solar.png",
                "description": "电离层扰动",
                "effects": {
                    "snr_penalty": -4,
                    "polar_boost": 2,  # Polar码在干扰下表现更好
                }
            }
        }

    def set_weather(self, weather_type):
        """设置天气"""
        if weather_type in self.weather_types:
            self.current_weather = weather_type
            return True
        return False

    def get_weather_info(self):
        """获取当前天气信息"""
        return self.weather_types[self.current_weather]

    def apply_weather_effects(self, base_snr, modulation, coding, use_laser=False):
        """应用天气效果"""
        weather = self.weather_types[self.current_weather]
        effects = weather["effects"]

        # 基础SNR惩罚
        snr = base_snr + effects.get("snr_penalty", 0)

        # 激光额外惩罚
        if use_laser:
            snr += effects.get("laser_penalty", 0)

        # Polar码在干扰下的增益
        if "Polar" in coding:
            snr += effects.get("polar_boost", 0)

        return snr
```

```python
                    "energy_cost": 1.2       # 能耗+20%
                },
                "best_for": "高级场景",
                "icon": "⚡",
                "unlock_level": 8  # 需要解锁
            }
        }

    def get_protocol_info(self, protocol_id):
        """获取协议信息"""
        return self.protocols.get(protocol_id, self.protocols["udp"])

    def apply_protocol_effects(self, protocol_id, base_ber, transmission_time, energy_cost):
        """应用协议效果"""
        protocol = self.protocols[protocol_id]
        effects = protocol["effects"]

        # 应用BER影响
        if "ber_improvement" in effects:
            final_ber = base_ber * effects["ber_improvement"]
        elif "ber_penalty" in effects:
            final_ber = base_ber * effects["ber_penalty"]
        else:
            final_ber = base_ber

        # 应用速度影响
        final_time = transmission_time / effects.get("speed_boost", 1.0)

        # 应用能耗影响
        if "energy_save" in effects:
            final_energy = energy_cost * effects["energy_save"]
        else:
            final_energy = energy_cost * effects.get("energy_cost", 1.0)

        return final_ber, final_time, final_energy
```

```python
def get_tech_score(modulation, coding, scenario_conditions):
    """
    计算技术组合在特定场景下的得分
    scenario_conditions: {"weather": "rain", "distance": "long", "snr": 5}
    """
    score = 100

    mod_char = TECH_CHARACTERISTICS[modulation]
    code_char = TECH_CHARACTERISTICS[coding]

    # 天气影响
    weather = scenario_conditions.get("weather", "clear")
    if weather in mod_char.get("weakness", []):
        score -= 20
    if weather in mod_char.get("best_scenarios", []):
        score += 15

    # SNR影响
    snr = scenario_conditions.get("snr", 10)
    if snr < 5 and coding == "Polar":
        score += 20
    if snr > 15 and coding == "None":
        score += 10

    return score
```

```python
class PowerSlider:
    def __init__(self, x, y, width, min_power=10, max_power=50):
        self.rect = pygame.Rect(x, y, width, 30)
        self.min_power = min_power
        self.max_power = max_power
        self.current_power = 30  # 默认30dBm
        self.dragging = False

    def handle_event(self, event):
        if event.type == pygame.MOUSEBUTTONDOWN:
            if self.rect.collidepoint(event.pos):
                self.dragging = True
        elif event.type == pygame.MOUSEBUTTONUP:
            self.dragging = False
        elif event.type == pygame.MOUSEMOTION and self.dragging:
            # 计算功率值
            rel_x = event.pos[0] - self.rect.x
            ratio = np.clip(rel_x / self.rect.width, 0, 1)
            self.current_power = self.min_power + ratio * (self.max_power - self.min_power)

    def draw(self, surface, budget_manager):
        # 滑块轨道
        pygame.draw.rect(surface, (60, 60, 60), self.rect)

        # 滑块位置
        ratio = (self.current_power - self.min_power) / (self.max_power - self.min_power)
        handle_x = self.rect.x + ratio * self.rect.width
        pygame.draw.circle(surface, (0, 180, 255), (int(handle_x), self.rect.centery), 15)

        # 功率值显示
        power_text = font.render(f"{self.current_power:.1f} dBm", True, TEXT_COLOR)
```

#### 2.3.2 分段传输

```python
# 修改传输逻辑，支持分段

class SegmentedTransmission:
    def __init__(self, data_bits, num_segments=5):
        self.segments = np.array_split(data_bits, num_segments)
        self.current_segment = 0
        self.results = []
        self.paused = False

    def send_next_segment(self, modulation, coding, snr):
        """发送下一段"""
        if self.current_segment >= len(self.segments):
            return None, True  # 全部完成

        segment = self.segments[self.current_segment]

        # 执行传输
        result = transmit_segment(segment, modulation, coding, snr)
        self.results.append(result)
        self.current_segment += 1

        return result, False

    def get_overall_ber(self):
        """计算总体BER"""
        total_bits = sum(len(seg) for seg in self.segments)
        total_errors = sum(r["errors"] for r in self.results)
        return total_errors / total_bits if total_bits > 0 else 0
```
