# 游戏可理解性与反馈改进方案

## 🎯 核心问题分析

### 当前问题

**玩家视角**：
- ❌ 不知道为什么赢了或输了
- ❌ 不清楚各个参数的实际影响
- ❌ BER、SNR等数值没有直观感受
- ❌ 感觉像在碰运气

**开发者视角**：
- ❌ 难以调整难度
- ❌ 不确定平衡性
- ❌ 无法预测玩家体验

### 根本原因

1. **反馈不足**：只有最终的BER结果，缺乏过程反馈
2. **因果关系模糊**：选择 → 结果的链条不清晰
3. **数值抽象**：技术指标对玩家来说太抽象
4. **缺乏对比**：没有参考系，不知道好坏
5. **随机性过高**：噪声的随机性让结果难以预测

class TransmissionPredictor:
    """传输结果预测器"""
    def __init__(self):
        pass

    def predict_outcome(self, modulation, coding, snr, weather, protocol):
        """预测传输结果"""
        # 计算理论BER
        theoretical_ber = self.calculate_theoretical_ber(modulation, coding, snr)

        # 应用天气影响
        weather_multiplier = get_tech_weather_multiplier(modulation, weather)
        adjusted_ber = theoretical_ber * weather_multiplier

        # 应用协议影响
        if protocol == "tcp":
            adjusted_ber *= 0.5
        elif protocol == "udp":
            adjusted_ber *= 1.2

        # 计算成功概率
        success_probability = self.calculate_success_rate(adjusted_ber, target_ber=0.01)

        # 计算预期传输时间
        transmission_time = self.calculate_time(modulation, coding, protocol)

        # 计算预期成本
        cost = calculate_transmission_cost(power_dbm=30, protocol=protocol)

        return {
            "predicted_ber": adjusted_ber,
            "success_probability": success_probability,
            "transmission_time": transmission_time,
            "cost": cost,
            "rating": self.get_rating(success_probability)
        }

    def calculate_success_rate(self, predicted_ber, target_ber):
        """计算成功概率（考虑随机性）"""
        if predicted_ber < target_ber * 0.5:
            return 0.95  # 很可能成功
        elif predicted_ber < target_ber:
            return 0.75  # 可能成功
        elif predicted_ber < target_ber * 1.5:
            return 0.50  # 不确定
        else:
            return 0.20  # 很可能失败

    def get_rating(self, success_probability):
        """获取评级"""
        if success_probability > 0.9:
            return "优秀", (0, 255, 0)
        elif success_probability > 0.7:
            return "良好", (100, 200, 100)
        elif success_probability > 0.5:
            return "一般", (255, 200, 0)
        else:
            return "较差", (255, 100, 100)
```

def draw_node(surface, x, y, text, color):
    """绘制节点"""
    pygame.draw.circle(surface, color, (x, y), 40)
    pygame.draw.circle(surface, (255, 255, 255), (x, y), 40, 2)

    lines = text.split('\n')

def draw_arrow(surface, x1, y1, x2, y2, value):
    """绘制箭头"""
    color = (0, 255, 0) if value >= 0 else (255, 100, 100)
    pygame.draw.line(surface, color, (x1, y1), (x2, y2), 3)

    # 箭头头部
    angle = np.arctan2(y2 - y1, x2 - x1)
    arrow_size = 10
    pygame.draw.polygon(surface, color, [
        (x2, y2),
        (x2 - arrow_size * np.cos(angle - 0.3), y2 - arrow_size * np.sin(angle - 0.3)),
        (x2 - arrow_size * np.cos(angle + 0.3), y2 - arrow_size * np.sin(angle + 0.3))
    ])

                "weight": 0.2,

            }
        ]

    def check_objective(self, obj_type, value):
        """检查目标完成情况"""
        for obj in self.objectives:
            if obj["type"] == obj_type:
                obj["current"] = value
                obj["completed"] = value <= obj["target"]

    def get_completion_rate(self):
        """获取完成度"""
        total_weight = sum(o["weight"] for o in self.objectives)
        completed_weight = sum(o["weight"] for o in self.objectives if o.get("completed", False))
        return completed_weight / total_weight
```

def channel_awgn_stable(symbols, snr_db, stability=0.8):
    """
    稳定的AWGN信道
    stability: 0-1，越高越稳定
    """
    # 原有随机噪声
    random_noise = np.random.randn(len(symbols))

    # 固定模式噪声
    fixed_noise = np.sin(np.arange(len(symbols)) * 0.1)

    # 混合
    noise = stability * fixed_noise + (1 - stability) * random_noise

    # 应用到信号
    noise_power = calculate_noise_power(snr_db)
    return symbols + noise * np.sqrt(noise_power)
```

**效果**：相同配置多次尝试，结果更一致

def predict_success(modulation, coding, snr, target_ber=0.01):
    """简化的成功预测"""
    # 计算理论BER（简化公式）
    if modulation == "BPSK":
        ber = 0.5 * np.exp(-snr)
    elif modulation == "QPSK":
        ber = 0.5 * np.exp(-snr * 0.7)
    else:
        ber = 0.5 * np.exp(-snr * 0.5)

    # 编码增益
    if "Polar" in coding:
        ber *= 0.3
    elif "Hamming" in coding:
        ber *= 0.5
    elif "Repetition" in coding:
        ber *= 0.4

    # 成功概率
    if ber < target_ber * 0.5:
        return "很可能成功", 0.9, (0, 255, 0)
    elif ber < target_ber:
        return "可能成功", 0.7, (100, 200, 100)
    elif ber < target_ber * 2:
        return "不确定", 0.5, (255, 200, 0)

        f"✓ 误码率 < {target_ber*100:.1f}%",
        f"✓ 时间 < {time_limit}秒",
        f"✓ 成本 < {budget_limit}"
    ]

    y = 200
    for obj in objectives:
        text = font.render(obj, True, TEXT_COLOR)
        surface.blit(text, (50, y))
        y += 30
```

2. **因果链** → 知道"为什么"

4. **教学模式** → 知道"怎么做"
5. **直观化** → 看得懂数值
6. **明确目标** → 知道要什么
7. **降低随机** → 结果可预测
